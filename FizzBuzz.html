function myFizzBuzz() {
var num1 = parseInt(document.getElementById("FirstNum").value);
var num2 = parseInt(document.getElementById("SecondNum").value);
    $("#placenumbers").html("");
    for (var i = 1; i <= 100; i++)
    {
        if (i % num1 == 0 && i % num2 == 0)
            $("#placenumbers").append(i + " FizzBuzz" + "
");
        else if (i % num1 == 0)
            $("#placenumbers").append(i + " Fizz" + "
");
        else if (i % num2 == 0)
            $("#placenumbers").append(i + " Buzz" + "
");
        else
            $("#placenumbers").append(i + "
");
    }
}
